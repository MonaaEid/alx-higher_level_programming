import unittest
from my_module import lazy_matrix_mul

class TestLazyMatrixMul(unittest.TestCase):
    def test_empty_matrices(self):
        self.assertRaises(ValueError, lazy_matrix_mul, [], [])
        self.assertRaises(ValueError, lazy_matrix_mul, [[]], [[]])

    def test_invalid_matrices(self):
        self.assertRaises(TypeError, lazy_matrix_mul, None, None)
        self.assertRaises(TypeError, lazy_matrix_mul, [], None)
        self.assertRaises(TypeError, lazy_matrix_mul, None, [])
        self.assertRaises(TypeError, lazy_matrix_mul, [[1], [2]], [3])
        self.assertRaises(TypeError, lazy_matrix_mul, [1], [[2], [3]])
        self.assertRaises(TypeError, lazy_matrix_mul, [[1], [2]], [3])
        self.assertRaises(ValueError, lazy_matrix_mul, [[1]], [[]])
        self.assertRaises(TypeError, lazy_matrix_mul, [[1], [2]], [[3], 4])

    def test_valid_matrices(self):
        self.assertEqual(lazy_matrix_mul([[1]], [[2]]), [[2]])
        self.assertEqual(lazy_matrix_mul([[1], [2]], [[3], [4]]), [[11]])
        self.assertEqual(lazy_matrix_mul([[1, 2], [3, 4]], [[5], [6]]), [[17], [39]])

if __name__ == '__main__':
    unittest.main()
